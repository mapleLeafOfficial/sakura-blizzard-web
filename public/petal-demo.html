① 真实樱花品种(预设)

真实樱花花瓣的关键特征:
  • 先端凹口(えぐれ)——樱属 Prunus 最具识别度的特征,一片心形凹缺;
  • 整体倒卵形,靠近凹口处最宽,向基部(花心着生点)收尖;
  • 常带轻微不对称;品种间在大小、长宽比、凹口深浅、颜色上差异明显。

// 6 个贝塞尔控制点定义花瓣轮廓(对应原版 _sakuraPetalV3d)
// front/back 两面用相同轮廓、不同 z,中间留 zDistance 形成厚度
function sakuraControlPoints(w, h, notchLen, z, front) {
  const L = -w/2, R = w/2, T = h/2, B = -h/2;
  const pts = [
    [  0, T - notchLen, z ],  // notchCenter 凹口底
    [ L/2,            T, z ],  // notchLeft
    [   L,            0, z ],  // mostLeft (贝塞尔控制点)
    [   0,            B, z ],  // bottomPoint 尖端
    [   R,            0, z ],  // mostRight (贝塞尔控制点)
    [ R/2,            T, z ],  // notchRight
  ];
  return front ? pts : pts.slice().reverse();
}

// 绘制一个面:二次贝塞尔轮廓 + camTheta 光照(最亮时 bri 可 >1,提升亮度)
function drawFace(ctx, p, camTheta, litBri, shadowBri, rgb) {
  const t = Math.max(0, Math.min(1, camTheta));
  const bri = shadowBri + (litBri - shadowBri) * t;   // 在 [shadowBri, litBri] 间插值
  const clip = v => Math.max(0, Math.min(255, Math.round(v * bri)));
  const col = `rgb(${clip(rgb[0])},${clip(rgb[1])},${clip(rgb[2])})`;
  ctx.beginPath();
  ctx.moveTo(p[0].x, p[0].y);                 // notchCenter
  ctx.lineTo(p[1].x, p[1].y);                 // → notchLeft
  ctx.quadraticCurveTo(p[2].x, p[2].y, p[3].x, p[3].y);  // 经 mostLeft → bottomPoint
  ctx.quadraticCurveTo(p[4].x, p[4].y, p[5].x, p[5].y);  // 经 mostRight → notchRight
  ctx.lineTo(p[0].x, p[0].y);                 // → notchCenter
  ctx.closePath();
  ctx.fillStyle = col;  ctx.fill();
  ctx.lineWidth = 1;    ctx.strokeStyle = col;  ctx.stroke();
}
每帧对每个面:用四元数旋转 6 个控制点 → 算面法线 → camTheta = 法线·视线 → 剔除背面(camTheta<0)→ 按深度排序 → drawFace 用最亮增益 + 阴影底之间的插值上色。
